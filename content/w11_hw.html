題目一:

已知可以用 Python 程式由上往下列印三角形 *, 或由左往右列印三角形 *, 請將此程式延伸到可以在 10x10 的字元區域中, 以 "*" 字元列印出圓型區域

第一題解答:"*"字號圓形

# 圓心位置
h, k = 5, 5
# 圓的半徑
r = 4
# 遍歷 10x10 的區域
for y in range(10):
    for x in range(10):
        # 計算每個點與圓心的距離
        distance = ((x - h) ** 2 + (y - k) ** 2) ** 0.5
        # 如果距離小於或等於圓的半徑，則列印 "*"
        if distance <= r:
            print('*', end='')
        else:
            print(' ', end='')  # 空格表示不是圓的一部分
    print()  # 換行到下一列

題目二:

已知可以在網頁中利用 Brython 繪製方格與塗色, 如下連結, 請設法修改此一程式, 直接將程式存入個人的 Gist, 並利用黑色方塊圍出一個圓形區域.網頁上的方格與塗色

第二題解答:網頁上的方格與塗色

from browser import html
from browser import document as doc
import math

# 利用 html 建立 canvas 超文件物件
canvas = html.CANVAS(width=400, height=400)
brython_div = doc["brython_div1"]
brython_div <= canvas

# 每一格的 pixel 數
gs = 20  # Grid size remains the same, each square is 20x20

# gs*tc = canvas width and height

ctx = canvas.getContext("2d")

def dRect(lux, luy, w, h, s=1, c='#ff0000'):
    ctx.lineWidth = s
    ctx.strokeStyle = c
    ctx.beginPath()
    ctx.rect(lux, luy, w, h)
    ctx.stroke()

def fill(x, y, color):
    ctx.fillStyle = color
    ctx.fillRect(x-1, y-1, gs, gs)

def drawFilledCircle(center_x, center_y, radius):
    # Fill the circle with black squares
    for i in range(center_x - radius, center_x + radius + 1):
        for j in range(center_y - radius, center_y + radius + 1):
            # Calculate the distance from the center
            distance = math.sqrt((i - center_x) ** 2 + (j - center_y) ** 2)
            
            # If the distance is less than or equal to the radius, fill the square
            if distance <= radius:
                fill(i * gs, j * gs, "black")  # Draw the square

def grid(width, height, grid_pix):
    for i in range(width):
        for j in range(height):
            dRect(i * grid_pix, j * grid_pix, grid_pix, grid_pix, 1, "lightgrey")

grid(10, 10, gs)  # Create a 10x10 grid
drawFilledCircle(5, 5, 4)  # Draw a filled circle with radius 4 at the center of the grid
